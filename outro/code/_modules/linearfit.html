Realizes a proper motion fit given the catalog produced by the friends of friends algorithm. This linear fit is simple and does not take into account correlated measurements.

Note that this file does a lot more than just linear fits. In fact, the most recent linear fit mechanic is currently implemented somewhere else. 
The big thing about this file is the Group class. This is a slow, but easy to use method of finding something with a given MATCH_ID.

May 2, 2017:
Created a new mechanic using the Group class to find the mean RA,DEC and compare to this position in the co-add catalog, thus giving the color. 

May 16, 2017:
Updated version of the color information, it uses a previously run FoF to match coadd color to finalcut.
Added a way to get the flux from the coadd and compare to the finalcut flux. Will change to magnitude soon


June 1, 2017:
Some of the old/less optimal versions of the code were removed. This file is only keeping the most up-to-date versions of each function, ready to use. Everything else was moved to the appropriate section in old.py
'''

[docs]def r(t, r0, v):
	'''
	Returns 
	 	:math:`\\mathbf{r} = \\mathbf{r}_0 + \\mathbf{v}_0 t`
	'''
	return r0 + v*t



[docs]def fit(x,y,t,sigma):
	'''Does a linear fit for r = r_0 + v*t, where r = (x,y) in gnomonic coordinates and t is the observation time. 
	Sigma is the error in each observation, given by the error in the ellipsis of the object, as measured by the telescope 
	Returns (x0, v_x), rho_x (cov matrix), (y0, v_y), rho_y '''
	x_params = curve_fit(r, t, x, sigma = sigma, absolute_sigma = True)
	y_params = curve_fit(r, t, y, sigma = sigma, absolute_sigma = True)

	return x_params[0], x_params[1], y_params[0], y_params[1]

[docs]	def get_coords(self):
		'''
		Gets RA, DEC, observation time and ERRAWIN_WORLD (error on the ellipsis) coordinates for all the detections in the group that are not COADD
		'''
		self.a_coords = [] #NEW_RA
		self.d_coords = [] #NEW_DEC
		self.p_errors = [] #ERRAWIN_WORLD
		self.time_obs = [] #MJD_OBS

		for i in self.points:
			if i['VERSION'] != "COADD":
				self.a_coords.append(i["NEW_RA"])
				self.d_coords.append(i["NEW_DEC"])
				self.p_errors.append(i["ERRAWIN_WORLD"]) # We'll soon add more errors here!
				self.time_obs.append(i["MJD_OBS"])



[docs]	def mean_pos(self):
		'''
		Returns the average RA and DEC of the group
		'''
		self.mean_ra = np.average(self.points['NEW_RA'])
		self.mean_dec = np.average(self.points['NEW_DEC'])


[docs]	def proper_motion(self):
		'''
		Finds the proper motion fit in a simple way (no assymetric errors) using scipy's linearfit
		'''
		self.get_coords()
		self.mean_pos()

		a_fit, a_cov, d_fit, d_cov = fit(self.a_coords, self.d_coords, self.time_obs, self.p_errors)
		self.fit_res = [a_fit[0], d_fit[0], np.sqrt(a_cov[0,0]), np.sqrt(d_cov[0,0]), a_fit[1], d_fit[1], np.sqrt(a_cov[1,1]), np.sqrt(d_cov[1,1]), a_cov[1,0], d_cov[1,0], self.group_id, self.mean_ra, self.mean_dec]
		self.a_fit = [a_fit, a_cov]
		self.d_fit = [d_fit, d_cov]

[docs]def get_ids(file_name):
	'''
	Gets all MATCH_IDs in the file
	'''
	table = tb.Table.read(file_name)
	id_list = []

	for i in range(int(max(table["MATCH_ID"])+1)):
		if list(table["MATCH_ID"]).count(i) > 2:
			id_list.append(i)

	return id_list




[docs]def create_catalog(file_name, output_file, plot=False):
	'''
	Computes the proper motion for all n_obs > 2 in a given file and returns a catalog of all the values
	'''
	
	#ids = get_ids(file_name)
	a = tb.Table.read(file_name)
	ids = np.unique(a['MATCH_ID'])

	groups = []
	for i in ids:
		group = Group(file_name, i)
		if len(group.points) > 2:
			group.proper_motion()
			if plot:
				group.plot()
			groups.append(group)


	new_cat = tb.Table(names=("RA_0", "DEC_0", "ERR_RA_0", "ERR_DEC_0","VEL_RA","VEL_DEC", "ERR_VEL_RA", "ERR_VEL_DEC", "COV_RA", "COV_DEC", "MATCH_ID", 'MEAN_RA', 'MEAN_DEC'))

	for i in groups:
		new_cat.add_row(i.fit_res)

	new_cat.write(output_file, overwrite=True)
	return groups




[docs]def add_color(file_name, output_file):
	''' 
	This is to be used after the first FoF run of the data! This should be used in place of create_color_catalog. This adds the color of each detection according to the coadd match found
	'''
	table = tb.Table.read(file_name)
	ids = np.unique(table['MATCH_ID'])

	avg_color = np.mean(table['COLOR'][np.where(table['VERSION'] == 'COADD')])
	colors = {}

	for i in ids:
		group = Group(table, i)
		if group.has_coadd():
			colors[i] = group.points['COLOR'][group.points['VERSION'] == 'COADD'][0]
		else:
			colors[i] = avg_color
	
	for detection in table:
		detection['COLOR'] = colors[detection['MATCH_ID']]

	table.write(output_file,overwrite=True)
