'''
Author: Pedro Bernardinelli
Original: Jul 27, 2017

This is a visualization tool for DES exposures (SINGLE EPOCH) and COADD images. It uses astropy's WCS features to use world coordinates in 
matplotlib's images, which is super convenient. One n by n image is made for single epoch, where n is round(sqrt(exposures in the region)) + 1
and 3x2 for coadds (as we have 5 images). We require the actual files for each image, which is pretty annoying...
The contrast level of the image is defined by Gary's clippedMean. If it fails, we try using astropy's ZScaleInterval.
We use the SCI extension of the FITS image, as we want.
'''

	'''
	Slow as hell way of finding the exposures in which the detections were made
	'''
	#tile = tb.Table.read('/home/pedro/Documents/DES/zone29/' + data['TILE'] + '_final.fits')
	match['EXPOSURES'] = '-'.join([str(i) for i in tile['EXPNUM'][tile['MATCH_ID'] == match['MATCH_ID']]])
	return match 


def exp_table(table):
	tile = tb.Table.read('/home/pedro/Documents/DES/zone29/' + table['TILE'][0] + '_final.fits')
	new_table = []
	for i in table:
		new_table.append(exp_data(i, tile))

	new_table = tb.vstack(new_table)

	return new_table


[docs]def get_exp(data, output):
	'''
	For a given MATCH_ID, finds all the exposures in which it appears and makes an EXPOSURES column that can be read easily by the other routines in this code 
	This is saved in the output file
	'''
	expo = tb.Column(name='EXPOSURES', dtype='U100', length = len(data))
	data.add_column(expo)

	tiles = ["DES0135-0416","DES0135-0458","DES0136-0541","DES0136-0624","DES0137-0707","DES0137-0250","DES0137-0333","DES0138-0416","DES0138-0458","DES0139-0541","DES0139-0624","DES0140-0250","DES0140-0707",
		"DES0140-0333","DES0140-0416","DES0141-0458","DES0141-0541","DES0142-0624","DES0143-0250","DES0143-0707","DES0143-0333","DES0143-0416","DES0144-0458","DES0144-0541","DES0145-0624","DES0146-0250","DES0146-0707",
		"DES0146-0333","DES0146-0416","DES0147-0458","DES0147-0541","DES0148-0624","DES0148-0250","DES0148-0707","DES0149-0333","DES0149-0416","DES0150-0458","DES0150-0541","DES0151-0624","DES0151-0250","DES0151-0707",
		"DES0152-0333","DES0152-0416","DES0152-0458","DES0153-0541","DES0154-0624","DES0154-0250","DES0154-0707"]
	
	new_data = []
	for i in tiles:
		new_data.append(data[data['TILE'] == i])
	

	pool = mp.Pool()
	expo = []
	expo.append(pool.map(exp_table, new_data)) 
	pool.close()

	data = tb.vstack(expo[0]) 
	data['RA'] = data['MEAN_RA']
	data['DEC'] = data['MEAN_DEC']
	data.write(output, overwrite=True)
	return data



[docs]def make_table(file_dir):
	'''
	Lists all single-epoch EXPNUM + CCDNUM and image type (image or segmap) for all files in a given directory
	'''
	os.chdir(file_dir)

	data = os.listdir(file_dir)
	exp = []
	ccd = []
	imtype = []


	for i in data[1:]:
		exp.append(int(i.split('_')[0][1:]))
		ccd.append(int(i.split('_')[2][1:]))
		imtype.append(i.split('_')[-1].split('.')[0])

	table = tb.Table()

	table['FILENAME'] = data[1:] 
	table['EXPOSURE'] = exp 
	table['CCD'] = ccd 
	table['TYPE'] = imtype

	table.write('list.fits')


def get_corners(file_dir):
	os.chdir(file_dir)
	a = tb.Table.read('list.fits')
	ra_min = []
	ra_max = [] 
	dec_min = [] 
	dec_max = []

	for i in a:
		try: 
			header = pf.getheader(i['FILENAME'], 'sci')
			ra_min.append(header['RACMIN'])
			ra_max.append(header['RACMAX'])
			dec_min.append(header['DECCMIN'])
			dec_max.append(header['DECCMAX'])
		except:
			ra_min.append(99999)
			dec_min.append(99999)
			ra_max.append(999999)
			dec_max.append(99999)

	a['RACMIN'] = ra_min
	a['RACMAX'] = ra_max
	a['DECCMIN'] = dec_min 
	a['DECCMAX'] = dec_max 
	a.write('list.fits', overwrite='True')



[docs]def find_exp_coords(ra, dec, file_dir, im_type):
	''' For all exposures, finds if the given RA, DEC is inside the exposure region. Returns immasked and segmaps OR coadd filenames for these images ''' 
	os.chdir(file_dir)
	a = tb.Table.read('list.fits')
	
	if im_type == 'single':
		im = []
		seg = [] 
		for i in a:
			if ra >= i['RACMIN'] and ra <= i['RACMAX']:
				if dec >= i['DECCMIN'] and dec <= i['DECCMAX']:
					if i['TYPE'] == 'segmap':
						seg.append(i['FILENAME'])
					elif i['TYPE'] == 'immasked':
						im.append(i['FILENAME'])

		return im, seg 

	if im_type == 'coadd':
		exp = []
		for i in a:
			if ra >= i['RACMIN'] and ra <= i['RACMAX']:
				if dec >= i['DECCMIN'] and dec <= i['DECCMAX']:
					exp.append(i['FILENAME'])
		return exp




[docs]def show_images_se(data_point, file_dir, folder, segmap = True, exposure_name=True):
	'''
	Single epoch visualization feature. For a given {RA, DEC}, it finds which exposures covered this part of the sky and generates the image centered around this point. 
	Also produces a 0.001 deg Spherical circle centered around this point, as the actual presented region is bigger than that. Images in which it was detected have a (D)
	in the title, and the circle is of a different color.
	It fails for some images and I don't know why.
	'''
	print 'Plotting SE'
	ra = data_point['RA']
	dec = data_point['DEC']
	im, seg = find_exp_coords(data_point['RA'], data_point['DEC'], file_dir, 'single')
	exposures = [int(i.split('_')[0][1:]) for i in im]
	fig = pl.figure(1)
	detected_exp = data_point['EXPOSURES'].split('-')
	n = int(np.sqrt(len(exposures)))
	for i in range(len(im)):
		fits = pf.open(im[i])
		image = fits['sci']
		print im[i]
		sigma = np.sqrt(cM(image.data, 4)[1])
		vmin = -3*sigma
		vmax = 4*sigma
				
		wcs = WCS(image.header)
		x,y = wcs.all_world2pix(ra,dec, 0)
		ax = fig.add_subplot(n+1,n+1,i+1,projection=wcs)
		ax.imshow(image.data, vmin=vmin, vmax=vmax, origin='lower', cmap='gray')
		ax.set_xlim([x - 75, x+75])
		ax.set_ylim([y - 75, y + 75])
		color = 'red'
		if exposure_name:
			if str(exposures[i]) in detected_exp:
				det = '(D)'
				color = 'cyan'
			else:
				det = ''
		else:
			det = ''

		ax.set_title('Exp: ' + str(exposures[i]) + ' Night: ' + image.header['NITE'] + " Filter: " + image.header['FILTER'][0] + ' ' + det, fontsize=10)
		r = SphericalCircle((ra * u.deg, dec * u.deg), 0.001 *u.deg, edgecolor=color, facecolor='none',transform=ax.get_transform('fk5'))
		ax.add_patch(r)

	#fig.suptitle('Match ID ' + str(data_point['MATCH_ID']) + ' for tile ' + str(data_point['TILE']) + r' where $\Delta t = $' + str(data_point['DELTA_T']) + r' and $\Delta m = $' + str(data_point['DELTA_M']))
	fig.set_size_inches(24, 24)
	fig.savefig(folder + str(data_point['TILE']) + '_' + str(data_point['MATCH_ID']) + '.pdf', dpi = 600)
	if segmap:

		fig = pl.figure(2)
		for i in range(len(im)):
			fits = pf.open(seg[i])
			image = fits['sci']
			wcs = WCS(image.header)
			x,y = wcs.all_world2pix(ra,dec, 0)
			ax = fig.add_subplot(n+1,n+1,i+1,projection=wcs)
			ax.imshow(image.data, origin='lower')
			ax.set_xlim([x - 75, x+75])
			ax.set_ylim([y - 75, y + 75])
			color = 'red'
			if str(exposures[i]) in detected_exp:
				det = '(D)'
				color = 'cyan'
			else:
				det = ''

			ax.set_title('Exp: ' + str(exposures[i]) + ' Night: ' + image.header['NITE'] + " Filter: " + image.header['FILTER'][0] + ' ' + det, fontsize=10)
			r = SphericalCircle((ra * u.deg, dec * u.deg), 0.001 *u.deg, edgecolor=color, facecolor='none',transform=ax.get_transform('fk5'))
			ax.add_patch(r)

		#fig.suptitle('Match ID ' + str(data_point['MATCH_ID']) + ' for tile ' + str(data_point['TILE']) + r' where $\Delta t = $' + str(data_point['DELTA_T']) + r' and $\Delta m = $' + str(data_point['DELTA_M']))

		fig.set_size_inches(24, 24)
		fig.savefig(folder + str(data_point['TILE']) + '_' + str(data_point['MATCH_ID']) + '_seg.pdf', dpi = 600)

	pl.close('all')

[docs]def show_images_coadd(data_point, file_dir, folder):
	'''
	Coadd visualization feature. For a given {RA, DEC}, it finds which tile it is part of and generates the image centered around this point. 
	Also produces a 0.001 deg Spherical circle centered around this point, as the actual presented region is bigger than that. 
	It fails for some images and I don't know why.

	'''
	print 'Plotting Coadd'
	ra = data_point['RA']
	dec = data_point['DEC']
	tiles = find_exp_coords(ra, dec, file_dir, 'coadd')

	fig = pl.figure(1)

	for i in range(len(tiles)):
		fits = pf.open(tiles[i])
		image = fits['sci']
		try: 
			sigma = np.sqrt(cM(image.data, 4)[1])
			vmin = -3*sigma
			vmax = 4*sigma
		except:
			vmin, vmax = ZScaleInterval().get_limits(image.data)

		wcs = WCS(image.header)
		x,y = wcs.all_world2pix(ra,dec, 0)
		ax = fig.add_subplot(3,2,i+1,projection=wcs)
		ax.imshow(image.data, vmin=vmin, vmax=vmax, origin='lower', cmap='gray')
		ax.set_xlim([x - 75, x+75])
		ax.set_ylim([y - 75, y + 75])
		
		ax.set_title('Coadd filter ' + image.header['BAND'][0] , fontsize=10)
		r = SphericalCircle((ra * u.deg, dec * u.deg), 0.001 *u.deg, edgecolor='red', facecolor='none',transform=ax.get_transform('fk5'))
		ax.add_patch(r)
		fits.close()

	#fig.suptitle('Match ID ' + str(data_point['MATCH_ID']) + ' for tile ' + str(data_point['TILE']) + r' where $\Delta t = $' + str(data_point['DELTA_T']) + r' and $\Delta m = $' + str(data_point['DELTA_M']))

	fig.set_size_inches(24, 24)
	fig.savefig(folder + str(data_point['TILE']) + '_' + str(data_point['MATCH_ID']) + '_COADD.pdf', dpi = 600)

	pl.close('all')
