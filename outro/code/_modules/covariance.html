'''
Author: Pedro Bernardinelli
May 1, 2017
Current: July 28, 2017

Calculates covariances from the exposures using Gary's astrometric files. This utility copies the data
using astroplots' select function and proceeds to obtain all the individual covariance matrices given the
residuals in v = (x,y) for a given weight factor.
Calculates the two-point correlation function 

:math:`\\xi(\\mathbf{r}) = \\langle \\Delta \\mathbf{x}(\\mathbf{x}) \\Delta \\mathbf{x}(\\mathbf{x}+\\mathbf{r}) \\rangle` 

for each exposure. From these values, we produce <xi>, the average of a given region,
so we can measure the total distortion in the image not coming from shot noise (xi(0)).
We also compute the ellipticity of the two-point function, where

:math:`e_1 = \\frac{I_{xx} - I{yy}}{I_{xx} + I_{yy}}`

:math:`e_2 = \\frac{2 I_{xy}}{I_{xx} + I_{yy}}`

July 28:
Clean-up and removing the treecorr files
'''

[docs]def get_data(fits, expo):
	'''
	Copies the data needed for both xi and sigma
	'''
	res = ap.select(fits,exposure=expo)

	xresW = res['xresW'] 
	yresW = res['yresW']
	sigW  = res['sigW']
	wtFrac = res['wtFrac']
	xW = res['xW']
	yW = res['yW']


	xv = xresW/(1-wtFrac)
	yv = yresW/(1-wtFrac)
	sig = 1./sigW*sigW

	return xv, yv, sig, xW, yW



[docs]def cov(xv, yv, sig):
	'''
	Calculates the covariance matrix for a given set of Delta x, Delta y and sigmas:

	:math:`\\boldsymbol{\\Sigma}_{ij} =  \\langle \\Delta x_i \\Delta x_j \\rangle`

	This is equivalent to :math:`\\xi(0)` for the exposure
	'''
	norm = np.sum(sig)/(np.sum(sig)**2. - np.sum(sig**2.))
	xx_cov = np.einsum('i,i,i', xv, xv, sig)*norm
	yy_cov = np.einsum('i,i,i', yv, yv, sig)*norm
	xy_cov = np.einsum('i,i,i', xv, yv, sig)*norm

	return xx_cov, yy_cov, xy_cov



[docs]def create_cov_file(fits, output_file):
	'''
	Catalog of covariances for each exposure
	'''	
	exposures = get_Expo(fits)
	cov_t = tb.Table(names=("EXPOSURE","XX_COV", "YY_COV", "XY_COV"), dtype=(np.int32,np.float64, np.float64,np.float64))

	for i in exposures:
		xv, yv, sig, xW, yW = get_data(fits, i)
		xx, yy, xy = cov(xv, yv, sig)
		if i[0] == "D":
			cov_t.add_row([i[1:], xx, yy, xy])
		else:
			cov_t.add_row([0, xx, yy, xy])
	cov_t.sort('EXPOSURE')
	cov_t.write(output_file, overwrite=True)



[docs]def integrate_array(r, xi, r_min, r_max):
	'''
	Computes 

	:math:`\\int_{r_\\mathrm{min}}^{r_\\mathbf{max}} \\mathrm{d}r \\, \\xi(r)`
	
	for a given function :math:`\\xi`. The array of this function must be indexed in the same manner as the r array
	'''
	r_interval = r[(r < r_max) & (r > r_min)]
	xi_interval = xi[(r < r_max) & (r > r_min)]

	return simps(xi_interval, r_interval)



[docs]def r_gauss(r):
	'''
	:math:`r \\cdot W(r)`, where :math:`W(r) = \\exp\\left(-\\frac{r^2}{2 (1\\text{arcsec})^2}\\right)`
	'''
	one_arc_min = 1./60
	return r*np.exp(-r**2/(2*(one_arc_min)**2.))

[docs]def xi_full(fits, min_r=5./3600, max_r=1.5, dlogr=0.05, order=3, keepnan = True):
	'''
	This computes all the two-point correlation functions for a given exposure.
	
	We first find a third order 2D polynomial that interpolates the data and subtract it from the data, to remove unwanted distortions
	Then, we proceed to find all the pairs of points in each bin. We compute, then, <xx>, <xy>, <yx> and <yy>, which allows us to get <xi_p>, <xi_x,y>, <xi_z> and Im(<xi_p>)
	After that, we rotate vv = xx - yy + i(xy + yx) into radial and perpendicular parts, so we can get the phase factor for <xi_m> and <xi_c>.
	Finally, from the analytical expressions, we compute <xi_E,B>. Returns all these functions plus log(r).
	'''
	#copying down the data and fitting the order 3 polynomials
	data = tb.Table.read(fits)
	x = np.asarray(data['X'])
	y = np.asarray(data['Y'])
	xp = ap.Poly2d(order, sumOrder=True)
	yp = ap.Poly2d(order, sumOrder=True)
	xp.fit(x,y,data['DELTA_X'])
	xfit = xp.evaluate(data["X"],data["Y"])
	yp.fit(x,y,data["DELTA_Y"])
	yfit = yp.evaluate(x,y)
	polyVar = np.var(xfit) + np.var(yfit)
	dx = np.asarray(data['DELTA_X']) - xfit
	dy = np.asarray(data['DELTA_Y']) - yfit
	
	points = range(len(dx))
	pairs = [i for i in combinations(points, 2)]
	pairs = np.asarray(pairs).T #gets all the unique combinations of the two points, excluding the i,i combinations

	dr = x[pairs[1]] - x[pairs[0]] + 1j*(y[pairs[1]] - y[pairs[0]]) #complex displacement vector

	logdr = np.log(np.absolute(dr))
	bins = int(np.ceil(np.log(max_r/min_r)/dlogr))
	hrange = (np.min(logdr), np.min(logdr)+bins*dlogr)
	counts = np.histogram(logdr, bins=bins, range=hrange)[0]
	logr = np.histogram(logdr, bins=bins, range=hrange, weights=logdr)[0]/counts

    #first let us compute xi_+
	v =  dx + 1j*dy #this is the v = dx + i dy part
	xx = dx[pairs[0]] * dx[pairs[1]] #dx dx correlation
	yy = dy[pairs[0]] * dy[pairs[1]] #dy dy corr
	xy = dx[pairs[0]] * dy[pairs[1]]
	yx = dy[pairs[0]] * dx[pairs[1]]
	vv = xx + yy + 1j*(xy - yx)#vv* correlation
	xi = np.histogram(logdr, bins=bins, range=hrange, weights=vv)[0]/counts
	xi_p = np.real(xi)
	xi_p_im = np.imag(xi) #this should be zero! =) 

	xi_x = np.histogram(logdr, bins=bins, range=hrange,weights=xx)[0]/counts
	xi_y = np.histogram(logdr, bins=bins, range=hrange, weights=yy)[0]/counts
	vv = xx - yy  + 1j*(xy + yx)
	xi_z = np.histogram(logdr, bins=bins, range=hrange, weights=vv)[0]/counts #vx1 vx2 - vy1 vy2 + 2i vx1 vy2 (actually, i(vx1 vy2 + vx2 vy1), but, if Im(xi_+) = 0, equivalent)

    # Now, xi_-
    #rotate vv into radial and perpendicular parts
	vv *= np.conj(dr)*np.conj(dr) # = dx^2 - dy^2 + 2ixy
	dr_abs = dr.real*dr.real + dr.imag*dr.imag
	vv /= dr_abs #effectively, we get a factor e^(-2itheta), where theta is the angle between i and j. This agrees with the definition of <vv exp(-2itheta)> of xi_-
	#the line above is the difference between the treecorr xi and this xi. In treecorr, we get exp(-i4theta), as the shear field is spin-2, unlike our vector field here
	xi_m = np.histogram(logdr, bins=bins, range=hrange, weights=vv)[0]/counts
	xi_c = np.imag(xi_m)
	xi_m = np.real(xi_m)

    #Now, for xi_E and xi_B
	dlogr = np.zeros_like(logr)
	dlogr[1:-1] = 0.5*(logr[2:] - logr[:-2]) 
	#as we only have logr, it's easier to integrate xi d(logr) than xi/r dr. Also, the cumulative sum function in numpy is really convenient here. Thanks, Gary!
	xir = xi_m*dlogr
	xi_int = np.nancumsum(xir[::-1])[::-1]
	xi_b = 0.5*(xi_p - xi_m) + xi_int
	xi_e = 0.5*(xi_p + xi_m) - xi_int
	if keepnan:
		return logr, xi_p, xi_p_im, xi_m, xi_c, xi_x, xi_y, xi_z, xi_e, xi_b
	else:
		return logr[np.isfinite(logr)], xi_p[np.isfinite(logr)], xi_p_im[np.isfinite(logr)], xi_m[np.isfinite(logr)], xi_c[np.isfinite(logr)], xi_x[np.isfinite(logr)], xi_y[np.isfinite(logr)], xi_z[np.isfinite(logr)], xi_e[np.isfinite(logr)], xi_b[np.isfinite(logr)] #those are lots of functions =)
