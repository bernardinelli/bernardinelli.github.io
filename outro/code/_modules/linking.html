[docs]def b_l(alpha, delta):
	'''
	For a given :math:`(\\alpha,\\delta)` (RA, DEC) measurement, finds the corresponding :math:`(b,\\ell)` angle
	'''
	b = np.arcsin(np.cos(epsilon) * np.sin(delta) - np.sin(epsilon) * np.cos(delta) * np.sin(alpha))
	l = np.arctan((np.cos(epsilon) * np.cos(delta) * np.sin(alpha) + np.sin(epsilon) * np.sin(delta))/(np.cos(delta) * np.cos(alpha)))
	return b,l



[docs]def theta_def(b, l, b0, l0):
	'''
	For a given :math:`(b,\\ell)` angle, finds the corresponding :math:`\\vec{\\theta}` angle 
	as measured according to a reference :math:`(b_0,\\ell_0)`
	'''
	den = np.sin(b0) * np.sin(b) - np.cos(b0) * np.cos(b) * np.cos(l - l0)
	theta_x = np.cos(b) * np.sin(l - l0)/ den
	den = np.sin(b0) * np.sin(b) + np.cos(b0) * np.cos(b) * np.cos(l - l0)
	theta_y = (np.cos(b0) * np.cos(b) - np.sin(b0) * np.sin(b) * np.cos(l - l0))/den
	return theta_x, theta_y

[docs]def earth_position(date, b0, l0):
	'''
	Uses the skyfield library to read a JPL ephemeris and find the position of the telescope according to a reference frame centered at the Sun and rotated
	by an angle :math:`(b_0, \\ell_0)`
	'''
	time =  time_skyfield.from_astropy(Time(date, format='mjd', scale='utc'))
	R_bl = np.array([[-np.sin(l0), np.cos(l0), 0],[-np.cos(l0)*np.sin(b0), -np.sin(l0)*np.sin(b0), np.cos(l0)],[np.cos(l0)*np.cos(b0), np.sin(l0)*np.cos(b0), np.sin(l0)]])

	return R_bl.dot(observer.at(time).position.au)





[docs]def theta_parallax(alpha, beta, gamma, r, r0):
	'''
	Calculates the parallax-only evolution of the :math:`\\theta` vector, defined by 

	:math:`\\theta_x(t) = \\frac{\\alpha - \\gamma x_E(t)}{1-\\gamma z_E(t)},`

	:math:`\\theta_y(t) = \\frac{\\beta  - \\gamma y_E(t)}{1-\\gamma z_E(t)}.`
	'''			
	r = r - r0
	theta_x = (alpha - gamma*r[0])/(1-gamma*r[2])
	theta_y = (beta  - gamma*r[1])/(1-gamma*r[2])
	return theta_x, theta_y



[docs]def theta_parallax_vec(theta, gamma, r, r0):
	'''
	Calculates the parallax-only evolution of the :math:`\\theta` vector, defined by 

	:math:`\\theta_x(t) = \\frac{\\alpha - \\gamma x_E(t)}{1-\\gamma z_E(t)},`

	:math:`\\theta_y(t) = \\frac{\\beta  - \\gamma y_E(t)}{1-\\gamma z_E(t)}.`

	This version is vectorized and accepts a vector as an input.
	'''			
	r = r - r0
	theta_n = (theta - gamma*r[:2])/(1 - gamma*r[2])
	return theta_n



[docs]def theta_real(alpha, beta, gamma, alpha_dot, beta_dot, gamma_dot, t, b, l, t0, r0):
	'''
	Calculates the full evolution of the :math:`\\theta` vector without the gravitational contribution, given by

	:math:`\\theta_x(t) = \\frac{\\alpha + \\dot{\\alpha} t- \\gamma x_E(t)}{1 + \\dot{\\gamma} t-\\gamma z_E(t)}`

	:math:`\\theta_y(t) = \\frac{\\beta  + \\dot{\\beta}  t- \\gamma y_E(t)}{1 + \\dot{\\gamma} t-\\gamma z_E(t)}`
	'''
	r = earth_position(t, b0, l0)

	t = (t - t0)/365.
	r = r -r0
	x = alpha + alpha_dot * t - gamma * r[0]
	y = beta + beta_dot * t - gamma * r[1]
	z = 1 + gamma_dot * t - gamma * r[2]
	return x/z, y/z





[docs]def theta_max(t1, t2, r, r0, gamma, b0, l0):
	'''
	Computes the maximum distance that an object at distance :math:`\\gamma` could have travelled between times 1 and 2
	'''	
	dt = (t2 - t1)/365.
	d_proper = 8*np.pi**2.*gamma**3.*dt**2.

	dr = r - r0

	phi = np.arctan(-(dr[2] - dr[0])/(dr[2] - dr[1]))

	d_parallax = gamma**3.*(np.linalg.norm(dr)**2. + dr[2]**2. - 2*dr[2]*(dr[0] + dr[1]))

	d_cross = 4*np.pi*np.power(gamma, 3./2)*dt*(np.cos(phi) * (dr[2] - dr[0]) + np.sin(phi)*(dr[2] - dr[1]))

	return np.sqrt(d_proper + d_parallax + d_cross)



[docs]def cov_dot(alpha, beta, gamma, dgamma, dt, Sigma_tp_1, Sigma_tp_2, r):
	'''
	For two measurements 1 and 2, with covariance matrices (in :math:`\\theta` space) :math:`\\Sigma_{1,2}`, finds the error matrix in 
	the estimated parameters :math:`\\alpha,\\beta,\\dot{\\alpha},\\dot{\\beta}`. 
	'''
	J = np.array([[1./(dt*r[2]*gamma - dt), 0, 1./dt, 0, (r[0] - alpha*r[2])/(dt*(-1 + r[2]*gamma)**2.) - r[0]/(1 - r[2] * gamma)], [0., 1./(dt*r[2]*gamma - dt), 0, 1./dt, (r[1] - beta*r[2])/(dt*(-1 + r[2]*gamma)**2.) - r[1]/(1 - r[2] * gamma)]])
	Sigma = np.bmat([[Sigma_tp_1, np.zeros_like(Sigma_tp_1)],[np.zeros_like(Sigma_tp_2),Sigma_tp_2]])
	Sigma = np.concatenate([Sigma, np.zeros((4,1))], axis=1)
	g_arr = np.array([[0,0,0,0,dgamma**2.]])
	Sigma = np.vstack((Sigma,g_arr))
	return np.dot(J, np.dot(Sigma, J.T))






[docs]def cov_theta(t1, t2, t3, alpha, beta, gamma, alpha_dot, beta_dot, dgamma, Sigma_tp_1, Sigma_tp_2, b0, l0, t0, r0):
	'''
	For two measurements 1 and 2, with covariance matrices (in :math:`\\theta` space) :math:`\\Sigma_{1,2}`, finds the region where a third measurement
	could be, defined as the propagation of the error matrix from the estimated parameters. That is, we first use 1 and 2 to find :math:`\\Sigma_{\\vec{\\alpha}}`
	and then use this matrix to find how it affects a measurement at a time t3
	'''
	dt = (t2 - t1)/365. 
	r = earth_position(t3, b0, l0)
	r2 = earth_position(t2, b0, l0)
	r = r - r0
	r2 = r2 - r0 
	t3 = (t3 - t0)/365. 

	J = np.array([[1./(1 - r[2] * gamma), 0, t3/(1. - r[2] * gamma), 0, r[2] * (alpha + t3 * alpha_dot - r[0]*gamma)/(1 - r[2] * gamma)**2.],[0, 1./(1 - r[2] * gamma), 0, t3/(1 - r[2] * gamma), r[2]*(beta + t3 * beta_dot - r[1] * gamma)/(1 - r[2] * gamma)**2.]])

	Sig_dot = cov_dot(alpha, beta, gamma, dgamma, dt, Sigma_tp_1, Sigma_tp_2, r2)
	g_arr = np.array([[0,0,0,0,dgamma**2.]])
	Sigma = np.bmat([[Sigma_tp_1, np.zeros_like(Sigma_tp_1)],[np.zeros_like(Sig_dot), Sig_dot]])
	Sigma = np.concatenate([Sigma, np.zeros((4,1))], axis=1)
	Sigma = np.vstack((Sigma,g_arr))
	return np.dot(J, np.dot(Sigma, J.T))



[docs]def search_radius(t1, t2, t3, gamma, Sigma_tp_1, Sigma_tp_2, b0, l0, t0, r0):
	'''
	Finds the semi-major axis of the ellipse defined by the covariance matrix estimation from cov_theta
	'''
	sigma = cov_theta(t1, t2, t3, gamma, Sigma_tp_1, Sigma_tp_2, b0, l0, t0)
	eig = np.linalg.eig(sigma)[0]
	a, b = eig[0], eig[1]
	return np.sqrt(a)

[docs]def alpha_est(theta_1, theta_2, t1, t2, gamma, t0, r0):
	'''
	Given two measurements :math:`\\vec{\\theta}_{1,2}`, finds the :math:`\\vec{\\alpha}` parameter that best describes it, for a fixed :math:`\\gamma`
	'''
	r1 = earth_position(t1, b0, l0)
	r2 = earth_position(t2, b0, l0)

	r1 = r1 - r0
	r2 = r2 - r0 
	t1 = (t1 - t0)/365.
	t2 = (t2 - t0)/365.

	M = np.array([[1, 0, t1, 0],[0, 1, 0, t1], [1, 0, t2, 0], [0, 1, 0, t2]])
	M_inv = np.linalg.inv(M)

	theta1_p = theta_1*(1 - gamma*r1[2]) + gamma*r1[:2]
	theta2_p = theta_2*(1 - gamma*r2[2]) + gamma*r2[:2]

	theta_v = np.bmat([theta1_p, theta2_p])

	params = M_inv.dot(theta_v.T)
	return params




[docs]def alpha_est_pair(pair, gamma, exposures):
	'''
	Pair version of the :math:`(\\alpha,\\beta,\\dot{\\alpha},\\dot{\\beta})` fit, given two measurements :math:`\\vec{\\theta}_{1,2}`, 
	finds the :math:`\\vec{\\alpha}` parameter that best describes it, for a fixed :math:`\\gamma`
	'''
	r1 = exposures[pair.det1.expnum].r
	r2 = exposures[pair.det2.expnum].r

	dr = r2 - r1
	dt = pair.det2.mjd - pair.det1.mjd
	M = np.array([[1, 0, 0, 0],[0, 1, 0, 0], [1, 0, dt, 0], [0, 1, 0, dt]])
	M_inv = np.linalg.inv(M)

	theta1_p = pair.det1.theta*(1 - gamma*dr[2])
	theta2_p = pair.det2.theta*(1 - gamma*dr[2])

	theta_v = np.bmat([theta1_p, theta2_p])

	params = M_inv.dot(theta_v.T)
	return params




[docs]def J_eq_to_ec(alpha, delta):
	'''
	Jacobian matrix of the :math:`(\\alpha,\\delta)` to :math:`(b,\\ell)` transformation
	'''
	den = np.sqrt(1 + (np.cos(epsilon) * np.sin(delta) - np.cos(delta) * np.sin(alpha) * np.sin(epsilon))**2.)
	J00 = -np.cos(alpha) * np.cos(delta) * np.sin(epsilon)/den
	J01 = (np.cos(delta) * np.cos(epsilon) + np.sin(alpha) * np.sin(delta) * np.sin(epsilon))/den
	den = 1 + (np.cos(epsilon) * np.tan(alpha) + (1./np.cos(alpha) * np.sin(epsilon) * np.tan(delta)))**2.
	J10 = (1./(np.cos(alpha)) * (np.cos(epsilon)/np.cos(alpha) + np.sin(epsilon) * np.tan(alpha) * np.tan(delta)))/den
	den = 1 + (np.cos(epsilon)**2. * np.tan(alpha))**2. + np.tan(alpha)/np.cos(alpha) * (np.sin(2*epsilon) * np.tan(alpha) + np.sin(epsilon)**2. * np.tan(delta)/np.cos(alpha))
	J11 = np.sin(epsilon)/(np.cos(alpha) * np.cos(delta)**2. * den)
	return np.array([[J00, J01],[J10, J11]])



[docs]def J_ec_to_tp(b, l, b0, l0):
	'''
	Jacobian matrix of the :math:`(b,\\ell)` to :math:`(\\theta_x,\\theta_y)` transformation, for a given reference :math:`(b_0, \\ell_0)`
	'''
	den = (np.cos(b0) * np.cos(l) * np.cos(b - l0) - np.sin(b0) * np.sin(l))**2.
	J00 = np.cos(l) * (-np.cos(b0) * np.cos(l) + np.cos(b-l0) * np.sin(b0) * np.sin(l))/den
	J01 = - np.sin(b0) * np.sin(b - l0)/den 
	den = (np.cos(b0) * np.cos(l) * np.cos(b - l0) + np.sin(b0) * np.sin(l))**2.
	J10 = (1 + np.cos(2* b0) * np.cos(2*l))*np.sin(b - l0)/(2*den)
	J11 = -(3 + np.cos(2*b - 2*l0))*np.sin(2*b0)/(4*den)
	return np.array([[J00, J01],[J10, J11]])

	:math:`\\cos\\beta_0 = -\\frac{(\\alpha, \\beta, 1) \\cdot \\mathbf{r}_0}{r_0 \\sqrt{\\alpha^2 + \\beta^2 + 1}}`
	'''
	r0 = -exposusres[pair.det1.expnum].r
	vec = np.array([pair.det1.x, pair.det1.y, 1])
	prod = np.inner(r0, vec)
	r0 = np.linalg.norm(r0)
	vec = np.linalg.norm(vec)

	return prod/(r0*vec)



[docs]def gamma_dot_bind(gamma, alpha_dot, beta_dot, cos_beta):
	'''
	Computes a limiting value for :math:`\\dot{\\gamma}`, given by 
	
	:math:`\\dot{\\gamma}_{\\text{bind}}^2 = 8 \\pi^2 \\gamma^3(1 + \\gamma^2 - 2 \\gamma \\cos\\beta_0)^{-1/2} - \\dot{\\alpha}^2 - \\dot{\\beta}^2`
	'''
	r = 1 + gamma**2. - 2 * cos_beta * gamma 
	gamma_bind_sq = 8 * np.pi**2. * gamma**3./np.sqrt(r) - alpha_dot**2 - beta_dot**2
	return np.sqrt(gamma_bind_sq)

		'''

		'''
		if self.kdtree is not None:
			res = self.kdtree.query_ball_point(point, radius)

[docs]	def covariance_matrix(self, exposures):
		'''
		Finds the error (exp[keys[i]].e. covariance) matrix from the position, according to the atmospheric part from the exposure
		'''
		#self.covariance  = (np.array([[self.pos_error**2., 0], [0, self.pos_error**2.]]) + exposures[self.expnum].cov) * (np.pi/180)**2
		self.correlation = (np.array([[self.pos_error**2., 0], [0, self.pos_error**2.]]) + exposures[self.expnum].corr) * (np.pi/180)**2

		J = J_eq_to_ec(self.ra * np.pi/180, self.dec * np.pi/180)
		Sigma_ec = np.dot(J,np.dot(self.correlation, J.T))
		G = J_ec_to_tp(self.b, self.l, b0, l0)
		Sigma = np.dot(G, np.dot(Sigma_ec, G.T))

		self.covariance = Sigma

[docs]	def alpha_est_pair(self, gamma, exposures):
		'''
		Pair version of the :math:`(\\alpha,\\beta,\\dot{\\alpha},\\dot{\\beta})` fit, given two measurements :math:`\\vec{\\theta}_{1,2}`, 
		finds the :math:`\\vec{\\alpha}` parameter that best describes it, for a fixed :math:`\\gamma`
		'''
		r1 = exposures[self.det1.expnum].r
		r2 = exposures[self.det2.expnum].r

		dr = r2 - r1
		dt = self.det2.mjd - self.det1.mjd
		M = np.array([[1, 0, 0, 0],[0, 1, 0, 0], [1, 0, dt, 0], [0, 1, 0, dt]])
		M_inv = np.linalg.inv(M)

		theta1_p = self.det1.theta*(1 - gamma*dr[2])
		theta2_p = self.det2.theta*(1 - gamma*dr[2])

		theta_v = np.bmat([theta1_p, theta2_p])

		params = M_inv.dot(theta_v.T)
		self.params = params
		return params



[docs]	def theta_pair(self, exposures, gamma, gamma_dot, r, t):
		'''
		Given a pair, a set of parameters :math:`(\\alpha,\\beta,\\dot{\\alpha},\\dot{\\beta})` coming from the pair fit, finds the position of a new detection at a 
		time :math:`t` with a fixed :math:`(\\gamma, \\dot{\\gamma})`. 
		'''
		t0 = self.det1.mjd
		r0 = exposures[self.det1.expnum].r

		t = (t - t0)/365.
		r = r -r0

		x = self.params[0] + self.params[2] * t - gamma * r[0]
		y = self.params[1] + self.params[3] * t - gamma * r[1]
		z = 1 + gamma_dot * t - gamma * r[2] 

		return x/z, y/z

[docs]class Candidate:
	'''
	Candidate class, consisting of a bunch of Detection objects added together. We can also make this work from the Pair class, or even from multiple Candidate objects.
	'''
	def __init__(self, *detections):
		'''
		We can input an arbitrary number of detections, with can be either Detections, Pairs or Candidates.
		'''	
		self.det = detections
		self.data = []
		for i in list(detections):
			if i.__class__ is Detection:
				self.data.append(i)
			elif i.__class__ is Pair:
				self.data.append(i.det1)
				self.data.append(i.det2)
			elif i.__class__ is Candidate:
				for j in i.data:
					self.data.append(j)
		self.data = np.unique(self.data)

	def __str__(self):
		'''
		Prints all Detections in the Candidate
		'''
		s = 'Candidate with detections: \n'
		for i in self.data:
			s = s + str(i) + '\n'
		return s[:-1]

		'''
