''' 
*Author: Pedro Bernardinelli*

*Original: Who knows? Sometime May 2017  - Current: July 28, 2017*

Split from :ref:`snap`. This is the graphical and plotting part of that code.

This program has the goal of transforming the second Friends of Friends output for one or multiple tiles into a catalog of detected objects, returning 
Delta t (defined in get_time), Delta m (defined coadd_mag), n_obs (in get_time), <RA> and <DEC> for all coadd-match objects and all but Delta m for the no-coadds.
As auxiliary information, we also preserve the MATCH_ID and TILE for each object.
'''

import numpy as np 
import astropy.table as tb
from snap import * 

import matplotlib.pyplot as pl 
pl.style.use('seaborn-paper')
from matplotlib import rc
rc('font',**{'family':'sans-serif','sans-serif':['Helvetica']})
rc('text', usetex=True)
rc('xtick.minor', visible = True, size=3)
rc('ytick.minor', visible = True, size=3)
rc('xtick.major', size=6)
rc('ytick.major', size=6)


[docs]def plot_histograms(tilename, filename, save=True,legend=False):
    '''
    This does everything that ``snap.obj_catalog`` does, but also plotting everything. Not useful at all. Can also skip saving the plots, so one could merge multiple regions into one.

    Also, this function is horribly long.
    '''
    distutils.dir_util.mkpath(tilename)

    a = tb.Table.read(filename)
    a.sort('MATCH_ID')
    groups = get_groups(a)

    #Let us first plot the Delta t histograms

    delta_t, n_obs, n_groups = time_cat(groups)
    pl.figure(1)
    if not legend:
        pl.suptitle('{:s} detections'.format(tilename))
    pl.subplot(231)
    pl.hist(delta_t,bins=50, histtype='step', label=tilename, linewidth=1.2)
    pl.ylabel('Counts', fontsize=13)
    pl.xlabel(r'$\max(\Delta t)$ (days)', fontsize=13)
    if legend:
        legend = pl.legend(loc=2, shadow=True,fontsize=11)
        frame = legend.get_frame()

    pl.subplot(232)
    pl.hist(delta_t[delta_t>0],bins=50, histtype='step', label=tilename, linewidth=1.2)
    pl.ylabel('Counts', fontsize=13)
    pl.xlabel(r'$\max(\Delta t) > 0$ (days)', fontsize=13)
    if legend:
        legend = pl.legend(loc=2, shadow=True,fontsize=11)
        frame = legend.get_frame()    
    
    pl.subplot(233)
    pl.hist(n_obs, bins=50, histtype='step', label=tilename, linewidth=1.2)
    pl.xlabel(r'$n_\mathrm{obs}$',fontsize=13)
    pl.ylabel('Counts', fontsize=13)
    if legend:
        legend = pl.legend(loc=0, shadow=True,fontsize=11)
        frame = legend.get_frame()

    pl.subplot(234)
    pl.hist(delta_t[n_obs>1],bins=50, histtype='step', label=tilename, linewidth=1.2)
    pl.ylabel('Counts', fontsize=13)
    pl.xlabel(r'$\max(\Delta t)$ (days) for $n_\mathrm{obs} > 1$', fontsize=13)
    if legend:
        legend = pl.legend(loc=2, shadow=True,fontsize=11)
        frame = legend.get_frame()

    pl.subplot(235)
    bins = np.linspace(0, 50, 50)
    pl.hist(24*delta_t[delta_t>0],bins=bins, histtype='step', label=tilename, linewidth=1.2)
    pl.ylabel('Counts', fontsize=13)
    pl.xlabel(r'$\max(\Delta t)$ (hours)', fontsize=13)
    if legend:
        legend = pl.legend(loc=0, shadow=True,fontsize=11)
        frame = legend.get_frame()


    pl.subplot(236)
    bins = np.linspace(0, 10, 20)
    pl.hist(24*60*delta_t[delta_t>0],bins=bins, histtype='step',label=tilename, linewidth=1.2)
    pl.ylabel('Counts', fontsize=13)
    pl.xlabel(r'$\max(\Delta t)$ (minutes)', fontsize=13)

    if legend:
        legend = pl.legend(loc=0, shadow=True,fontsize=11)
        frame = legend.get_frame()

    fig = pl.gcf()    
    fig.set_size_inches(18.5, 10.5)

    if save:
        pl.savefig(tilename + '/' +tilename+'_time.pdf')

    # Now for the magnitude plots!

    pl.figure(2)
    if not legend:
        pl.suptitle('{:s} magnitude differences'.format(tilename))

    coadd_groups = a['MATCH_ID'][a['VERSION'] == 'COADD']

    mag = mag_cat(coadd_groups, groups)

    g = []
    r = []
    i = []
    z = []
    Y = []

    for v in mag.values():
        for j in v:
            g.append(j['g'])
            r.append(j['r'])
            i.append(j['i'])
            z.append(j['z'])
            Y.append(j['Y'])

    g_mag = []
    r_mag = []
    i_mag = []
    z_mag = []
    Y_mag = []
    min_mag = []
    del_t = []

    c_groups = np.asarray(coadd_groups)
    ids = np.intersect1d(n_groups, c_groups)

    for i in ids:
        g_mag.append(np.min([j['g'] for j in mag[i]]))
        r_mag.append(np.min([j['r'] for j in mag[i]]))
        i_mag.append(np.min([j['i'] for j in mag[i]]))
        z_mag.append(np.min([j['z'] for j in mag[i]]))
        Y_mag.append(np.min([j['Y'] for j in mag[i]]))
        min_mag.append(np.min([j.values() for j in mag[i]]))
    min_mag = np.asarray(min_mag)
    bins = np.linspace(-3, 3, 50)

    index = [i for i in range(len(delta_t)) if n_groups[i] in ids]
    t_coadd = delta_t[index]

    pl.subplot(221)
    pl.hist(min_mag, bins=bins, histtype='step', label=tilename, linewidth=1.2)
    pl.xlabel(r'$\min(\Delta m)$ (mag)',fontsize=11)
    pl.ylabel('Counts',fontsize=11)

    if legend:
        legend = pl.legend(loc=0, shadow=True,fontsize=11)
        frame = legend.get_frame()


    
    pl.subplot(222)
    pl.hist(g, histtype='step', bins=bins, color='g', label="g", linewidth=1.2)
    pl.hist(r, histtype='step', bins=bins, color='r', label="r", linewidth=1.2)
    pl.hist(i, histtype='step', bins=bins, color='m', label="i", linewidth=1.2)
    pl.hist(z, histtype='step', bins=bins, color='y', label="z", linewidth=1.2)
    pl.hist(Y, histtype='step', bins=bins, color='k', label="Y", linewidth=1.2)
    pl.legend()
    pl.xlabel(r'$\min(\Delta m)$ (mag)',fontsize=11)
    pl.ylabel('Counts',fontsize=11)

    pl.subplot(223)

    pl.hist(min_mag[t_coadd > 0], bins=bins, histtype='step', label=tilename, linewidth=1.2)
    if legend:
        legend = pl.legend(loc=0, shadow=True,fontsize=10)
        frame = legend.get_frame()

    pl.xlabel(r'$\min(\Delta m)$ (mag) for $\Delta t > 0$',fontsize=11)
    pl.ylabel('Counts',fontsize=11)      

    pl.subplot(224)
    index = [i for i in range(len(delta_t)) if n_groups[i] in ids]
    t_coadd = delta_t[index]

    pl.hist(min_mag[t_coadd < 2], bins=bins, histtype='step', label=tilename, linewidth=1.2)
    pl.xlabel(r'$\min(\Delta m)$ (mag)',fontsize=11)
    pl.ylabel('Counts',fontsize=11)

    if legend:
        legend = pl.legend(loc=0, shadow=True,fontsize=10)
        frame = legend.get_frame()

    pl.xlabel(r'$\min(\Delta m)$ (mag) for $\Delta t < 2$ days',fontsize=11)
    pl.ylabel('Counts',fontsize=11)

    fig = pl.gcf()    
    fig.set_size_inches(12, 8)

    if save:
        pl.savefig(tilename + '/' +tilename+'_mag.pdf')

    #output file and Delta t vs Delta mag distribution

    pl.figure(3)

    if not legend:
        pl.suptitle('{:s} detections and magnitudes'.format(tilename))

    index = [i for i in range(len(delta_t)) if n_groups[i] in ids]
    t_coadd = delta_t[index]
    mean_ra  = []
    mean_dec = []
    for i in index:
        mean_ra.append(np.average([d['NEW_RA'] for d in groups[i]]))
        mean_dec.append(np.average([d['NEW_DEC'] for d in groups[i]]))


    pl.plot(24*60*t_coadd, np.asarray(min_mag),'.', label=tilename)
    pl.xlabel(r'$\max(\Delta t)$ (minutes)', fontsize=13)
    pl.ylabel(r'$\min(\Delta m)$ (mag) for $\Delta t < 2$ days', fontsize=13)
    if legend:
        legend = pl.legend(loc=0, shadow=True,fontsize=10)
        frame = legend.get_frame()
    pl.xlim([-1,10])

    if save:
        pl.savefig(tilename + '/' +tilename+'_combined.pdf')

    output = tb.Table((index,t_coadd, mean_ra, mean_dec, min_mag),names=("MATCH_ID","DELTA_T","MEAN_RA","MEAN_DEC","DELTA_M"))

    output.write(tilename+'/'+tilename+"_out.csv")

[docs]def hist_detections(coadd, nocoadd, borders):
    '''
    Keeping for historic/debugging purposes, sees how the edges of the tile affects the number of no-coadds
    '''
    n = 50

    coadd = tb.Table.read(coadd)
    nocoadd = tb.Table.read(nocoadd) 
    borders = tb.Table.read(borders)
    cuts = [0, 0.0001, 0.0003, 0.005, 0.01, 0.02, 0.05, 0.07, 0.1, 0.2, 0.5]
    ones = []
    notones = []
    everything = []
    for dr in cuts:
        new_no_coadd = nocoadd
        for i in borders['MIN_DEC']:
            new_no_coadd = new_no_coadd[~np.logical_and(new_no_coadd['MEAN_DEC'] < i + dr, new_no_coadd['MEAN_DEC'] > i-dr)]
        for i in borders['MAX_DEC']:
            new_no_coadd = new_no_coadd[~np.logical_and(new_no_coadd['MEAN_DEC'] < i + dr, new_no_coadd['MEAN_DEC'] > i-dr)]
        ones.append(len(new_no_coadd[new_no_coadd['N'] == 1]))
        notones.append(len(new_no_coadd[new_no_coadd['N'] != 1]))
        everything.append(len(new_no_coadd))
        pl.hist(new_no_coadd['N'], bins=50, histtype='step', linewidth=1.4, label='Cut = '+str(dr))

    #pl.yscale('log')
    pl.xlabel(r'$n_\mathrm{obs}$', fontsize=12)
    pl.ylabel('Counts', fontsize=12)
    pl.title('No-coadd matches with cuts on the edges of each tile')
    legend = pl.legend(loc=0, shadow=True,fontsize=10)
    frame = legend.get_frame()
    pl.savefig('n_det_no_coadd.pdf')
    pl.cla()

    ra = np.linspace(np.min(nocoadd['MEAN_RA']), np.max(nocoadd['MEAN_RA']), 2000)
    dec = np.linspace(np.min(nocoadd['MEAN_DEC']), np.max(nocoadd['MEAN_DEC']), 2000)


    for i in range(7):
        pl.plot([min(ra), max(ra)],[min(dec) + (max(dec)-min(dec))/7*i, min(dec) + (max(dec)-min(dec))/7*i], 'k--', linewidth=3, alpha=0.6)
        pl.plot([min(ra) + (max(ra)-min(ra))/7*i, min(ra)+ (max(ra)-min(ra))/7*i],[min(dec), max(dec)], 'k--', linewidth=3, alpha=0.6)

    new_no_coadd = nocoadd
    dr = 0.1
    for i in borders['MIN_DEC']:
        new_no_coadd = new_no_coadd[~np.logical_and(new_no_coadd['MEAN_DEC'] < i + dr, new_no_coadd['MEAN_DEC'] > i-dr)]
    for i in borders['MAX_DEC']:
        new_no_coadd = new_no_coadd[~np.logical_and(new_no_coadd['MEAN_DEC'] < i + dr, new_no_coadd['MEAN_DEC'] > i-dr)]
     
    
    new_no_coadd = new_no_coadd[new_no_coadd['N'] == 1]   
    pl.hist2d(new_no_coadd['MEAN_RA'], new_no_coadd['MEAN_DEC'],bins=200)

    pl.xlabel(r"$\alpha$", fontsize=12)
    pl.ylabel(r'$\delta$', fontsize=12)
    pl.title('Coadd match distribution for zone 29')
    fig = pl.gcf()    
    fig.set_size_inches(12, 8)
    pl.savefig('nocoadd_clipping.pdf')

    pl.cla()
    pl.plot(cuts, ones, label=r'$n_\mathrm{obs} = 1$')
    pl.plot(cuts, notones, label=r'$n_\mathrm{obs} > 1$')
    pl.plot(cuts, everything, label='Both')
    pl.axvline(x = 0.0003, color='k', linestyle=':', alpha=0.2)
    pl.axvline(x=0.05, color='k', linestyle=':', alpha=0.4)
    pl.xscale('log')
    pl.xlabel(r'$\delta$ cut (deg)', fontsize=12)
    pl.ylabel('Counts', fontsize=12)
    pl.title('Number of coadd matches after cutting the edges of each tile on DEC', fontsize=15)
    legend = pl.legend(loc=0, shadow=False,fontsize=12, frameon = False)
    frame = legend.get_frame()
    fig = pl.gcf()    
    fig.set_size_inches(12, 8)
    pl.savefig('nocoadd_dist.pdf')
