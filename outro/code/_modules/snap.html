[docs]def get_groups(table):
    '''For a given data table (i.e. an output from the second FoF) sorted by MATCH_ID, creates a dictionary indexed by the MATCH_ID where each item is a list of every appearance of this ID in table. 
    vstacking would be ideal, as, then, each group would have its own table, but this process is time consuming and slows the process significantly '''
    groups = {}
    i = 0
    id_n = 0
    groups[0] = []
    while i < len(table):
        if table['MATCH_ID'][i] == id_n:
            groups[id_n].append(table[i])
            i += 1
        else:
            #groups[id_n] = tb.vstack(groups[id_n])
            id_n +=1
            groups[id_n] = [table[i]]
            i += 1
    return groups



[docs]def get_time(data):
    ''' For a given data list (i.e. an entry in get_groups' dictionary), finds 

    :math:`\\Delta t \\equiv \\max(|t_i - t_j|)`, 

    for all i,j observations of the same object. The function returns :math:`\\Delta t`, the second maximum value and 

    :math:`n_\\mathrm{obs} \\equiv \\text{number of single-epoch detections of the given object}.`

    For an object with only one single-epoch detection, we get {0,0,1}. For the coadd-only objects (that is, the objects with no single epoch detection), we get {-1,-1,0}.
    '''
    times = np.asarray([i['MJD_OBS'] for i in data if i['VERSION'] == 'v2.0'])
    diff = []
    if len(times)  > 1:
        for i in range(len(times)):
            for j in range(0,i):
                   diff.append(np.abs(times[i]-times[j]))
        diff = np.sort(np.asarray(diff))

        try:
            return np.max(diff), diff[-2], len(times)
        except:
            return np.max(diff), np.max(diff), len(times)
    elif len(times) == 1:
        return 0, 0, 1
    else:
        return -1, -1, 0

   

[docs]def time_cat(table):
    '''
    For a given get_groups dictionary, returns :math:`\\Delta t`, :math:`\\Delta t_2`, :math:`n_\\mathrm{obs}` and MATCH_ID for all the entries in this dic.
    Note that, as every object in the catalog has an associated Delta t and appears in the dictionary, n_group is isomorphich to MATCH_ID.  
    This was not true in the previous versions of the code and led to a mistake in the identification of no-coadds
    '''
    times = []
    times2 = []
    n_obs = []
    n_group = []
    for i in range(len(table)):
        t, t2, n = get_time(table[i])
        if np.isfinite(t):
            times.append(t)
            times2.append(t2)
            n_obs.append(n)
            n_group.append(i)
            
    return np.asarray(times), np.asarray(times2), np.asarray(n_obs), np.asarray(n_group)     



[docs]def coadd_mag(groups):
    '''For a given COADD-match list of catalog appearances, finds all the coadd magnitudes and proceeds to compute 

    :math:`\\Delta m_f \\equiv \\min_{\\mathrm{abs}} (m_{\\text{single epoch},i,f} - m_{\\text{coadd},f}),`

    where f is the filter, i is the single-epoch detection and min_abs is the sign-preserving absolute minimum, 
    that is, the closest value to zero, indiscriminant of sign, is kept, but we preserve its sign. The function returns Delta m_f for all the filters. 
    Note that if the coadd doesn't have the magnitude for a certain filter, we'll have m = +- 99, so this value is going to be huge
    '''
    coadd_mag = []
    for i in groups:
        if i['VERSION'] == 'COADD':
            coadd_mag.append({'g': i['MAG_AUTO_G'], 'r': i['MAG_AUTO_R'], 'i' : i['MAG_AUTO_I'], 'z': i['MAG_AUTO_Z'], 'Y': i['MAG_AUTO_Y']})

    delta_mag = [{'g': 99, 'r': 99, 'i': 99, 'z': 99, 'Y': 99} for i in range(len(coadd_mag))]
    for i in groups:
        if i['VERSION'] != 'COADD':
            mag = -2.5*np.log10(i['FLUX_AUTO']) + i['MAG_ZERO']
            for j in range(len(coadd_mag)):
                if np.abs(mag - coadd_mag[j][i['BAND']]) < np.abs(delta_mag[j][i['BAND']]):
                    delta_mag[j][i['BAND']] = mag - coadd_mag[j][i['BAND']]
    return delta_mag

[docs]def obj_catalog(tilename, filename):
    '''
    For a given tile(name) and FoF-output (filename), finds :math:`\\Delta t`, :math:`\\Delta m \\equiv \\min_{\\mathrm{abs}}(\\Delta m_f)`, :math:`n`, 
    :math:`\\langle \\alpha \\rangle` and :math:`\\langle \\delta \\rangle` for all detected objects,
    returning them in two files: one for all coadd matches and all for all no-coadd matches.
    This started as the non-graphical version of plot_histograms (now in snap_plots)
    '''
    a = tb.Table.read(filename)
    #making sure that the table is sorted by MATCH_ID, as required by get_groups
    a.sort('MATCH_ID')
    groups = get_groups(a)

    #time catalog assigns Delta_t = -1 to things that do NOT have a single epoch, but do have a coadd exposure.
    #note that here we do not discriminate between coadds and non-coadds
    delta_t, delta_t2, n_obs, n_groups = time_cat(groups) 

    #coadd-only
    coadd_groups = np.unique(a['MATCH_ID'][a['VERSION'] == 'COADD'])
    #if something's magnitude is not a coadd match for some reason, we get a +- 99 
    mag = mag_cat(coadd_groups, groups)

    min_mag = []
    #finds which coadd groups are part of n_groups
    ids = np.intersect1d(np.array(coadd_groups), np.array(n_groups))

    for i in ids:
        data = np.asarray([j.values() for j in mag[i]]).flatten()
        min_mag.append(data[np.argmin(np.abs(data))])

    min_mag = np.asarray(min_mag)
    #indexing so we can cut the non-coadds off
    index = [i for i in range(len(delta_t)) if n_groups[i] in coadd_groups]
    n_coadd = n_obs[index]
    t_coadd = delta_t[index]
    t_coadd2 = delta_t2[index]
    mean_ra  = []
    mean_dec = []
    for i in index:
        mean_ra.append(np.average([d['NEW_RA'] for d in groups[i]]))
        mean_dec.append(np.average([d['NEW_DEC'] for d in groups[i]]))

    #this is the coadd output file
    output = tb.Table((index,t_coadd, t_coadd2, mean_ra, mean_dec, min_mag, n_coadd),names=("MATCH_ID","DELTA_T", 'DELTA_T2', "MEAN_RA","MEAN_DEC","DELTA_M", "N"))
    output.write('zone29/'+tilename+"_out_coadd.fits", overwrite=True)

    #everything that is not on the coadd ends up here. There's, of course, no magnitude match. 
    index = [i for i in range(len(delta_t)) if n_groups[i] not in coadd_groups]
    t_non_coadd = delta_t[index]
    t_non_coadd2 = delta_t2[index]
    n_non_coadd = n_obs[index]

    mean_ra  = []
    mean_dec = []
    for i in index:
        mean_ra.append(np.average([d['NEW_RA'] for d in groups[i]]))
        mean_dec.append(np.average([d['NEW_DEC'] for d in groups[i]]))

    #No-coadd output file
    output = tb.Table((index,t_non_coadd, t_non_coadd2, mean_ra, mean_dec, n_non_coadd),names=("MATCH_ID","DELTA_T", "DELTA_T2", "MEAN_RA","MEAN_DEC", "N"))
    output.write('zone29/'+tilename+"_out_nocoadd.fits", overwrite=True)

[docs]def tile_cat(tiles, nodes):
    '''
    This script generates a coadd and a noncoadd catalog for each of the tiles, using the given number of nodes
    '''
    files = []
    for i in tiles:
        files.append('/Users/pedro/Documents/DES/zone29/'  + i + '_final.fits')

    iterable = it.izip(tiles, files)

    pool = mp.Pool(processes = nodes)
    pool.map(obj_star,iterable) #list of each projected WCS Table rows 
        
    pool.close()
